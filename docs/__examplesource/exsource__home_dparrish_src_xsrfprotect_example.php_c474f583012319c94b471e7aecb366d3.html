Sample usage

  1. <?php
  2.  
  3. include "XsrfProtection.php";
  4.  
  5. // Create the protection object
  6. $prot = new XsrfProtection();
  7. // Set a secret key. This should really be secret, but must be shared between
  8. // all your frontends that will be handling this type of request.
  9. $prot->SetKey("CHANGE_THIS_KEY_TO_SOMETHING_ONLY_YOU_KNOW");
  10. // Set some user data. This would normally be the logged in user or some other
  11. // identifying field.
  12. $prot->SetUserData("dparrish");
  13. // Set the URL field
  14. $prot->SetUrl("http://www.test.com/xsrf/example.php");
  15. // Set the maximum age of tokens. The default is 1 hour.
  16. $prot->SetTimeout(3600);
  17. // Use the session cache to prevent replay attacks.
  18. session_start();
  19. $prot->SetStateful();
  20.  
  21.  
  22. if (isset($_REQUEST['test'])) {
  23.   // Validate the submitted form token.
  24.   $ret = $prot->Validate($_POST);
  25.   if ($ret != XsrfProtection::kCheckSuccess) {
  26.     // There was an error, print out the reason. This is not a good idea in
  27.     // production, but works for an example.
  28.     print "XSRF detected, validation failure ". $prot->Error(). "\n";
  29.     return;
  30.   }
  31.   // All good, print something out and return
  32.   print "Test value: ". $_REQUEST['test'];
  33.   return;
  34. }
  35.  
  36. // Generate a very basic form that has XSRF protection.
  37. ?>
  38. <form method='post'>
  39. <input type='text' name='test' value='foobar'>
  40. <input type='submit' name='doit' value='Submit'>
  41. <?=$prot->ProtectionField()?>
  42. </form>
